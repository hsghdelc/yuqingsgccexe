// 渲染实体词库强弱高亮背板：逐行按规则上色，保持与 textarea 完全对齐
function renderEntityBackdrop() {
  const ta = $('entityTextarea');
  const bd = $('entityBackdrop');
  if (!ta || !bd) return;
  const lines = ta.value.split('\n');
  let html = '';
  for (let i = 0; i < lines.length; i++) {
    const raw = lines[i];
    const word = raw.trim();
    if (word) {
      const cls = isStrongEntity(word) ? 'ent-strong' : 'ent-weak';
      // 保留行内前后空白，仅给词本身上色
      const lead = raw.slice(0, raw.indexOf(word));
      const tail = raw.slice(raw.indexOf(word) + word.length);
      html += escHtml(lead) + `<span class="${cls}">` + escHtml(word) + '</span>' + escHtml(tail);
    } else {
      html += escHtml(raw);
    }
    if (i < lines.length - 1) html += '\n';
  }
  // 末尾补一个换行，避免最后一行裁切；与 textarea 行为一致
  bd.innerHTML = html + '\n';
  bd.scrollTop = ta.scrollTop;
  bd.scrollLeft = ta.scrollLeft;
}

    $('strongCount').textContent = strong;
    $('weakCount').textContent = entities.length - strong;
  }
  renderEntityBackdrop();
}

function loadFile(file) {
  if (!file.name.endsWith('.xlsx')) { alert('仅支持 .xlsx 格式文件'); return; }
  originalFileName = file.name.replace(/\.xlsx$/i, '');
  const reader = new FileReader();
  reader.onload = ev => {
    try {
      window._origArrayBuffer = ev.target.result.slice(0);
      wb = XLSX.read(new Uint8Array(ev.target.result), { type: 'array', cellStyles: true });
      log(`文件已加载: ${file.name}（${(file.size / 1024).toFixed(1)} KB）`, 'ok');
      $('fileInfoName').textContent = file.name;

      const sheets = wb.SheetNames;
      const ws = wb.Sheets[sheets[0]];
      const range = XLSX.utils.decode_range(ws['!ref'] || 'A1');
      const rowCount = range.e.r; // 不含标题
      $('fileInfoMeta').textContent = `${rowCount} 条数据 · ${sheets.length} 个工作表`;
      $('fileInfoBlock').style.display = '';

      // 多Sheet选择
      const selectWrap = $('sheetSelectWrap');
      const sel = $('sheetSelect');
      sel.innerHTML = '';
      sheets.forEach(s => {
        const opt = document.createElement('option');
        opt.value = s; opt.textContent = s;
        sel.appendChild(opt);
      });
      currentSheetName = sheets[0];
      selectWrap.style.display = sheets.length > 1 ? '' : 'none';

      // 抓取表头并渲染字段映射下拉框
      const rawDataForHeaders = XLSX.utils.sheet_to_json(ws, { header: 1, defval: '' });
      window.currentExcelHeaders = rawDataForHeaders.length > 0 ? rawDataForHeaders[0].map(h => String(h || '').trim()) : [];
      renderMappingDropdowns(6); // 默认渲染 6 个，刚好凑成 3 行 2 列
      populateExtractDropdown(); // 填充内容解析下拉框

      $('startBtn').disabled = false;
      setStatus('文件已加载', 'info');
      resultRows = [];
      renderPreview([]);
    } catch (err) {
      log('文件解析失败: ' + err.message, 'err');
    }
  };
  reader.readAsArrayBuffer(file);
}

$('sheetSelect').addEventListener('change', e => {
  currentSheetName = e.target.value;
  log(`已切换工作表: ${currentSheetName}`, 'info');
});

      entity: $('entityTextarea').value,
      domain: $('domainTextarea').value,
      neg: $('negTextarea').value,
      exempt: $('exemptTextarea').value
    }
  };

  // 保存当前模型的API Key
  models.forEach(m => {
    const keyInput = $('apiKeyInput');
    if (m.id === activeModelId && keyInput) {
      config.apiKeys[m.id] = keyInput.value.trim();
    }
  });

  // 保存额外的Key
  const extraInput = $('extraKeysInput');
  if (extraInput) {
    config.extraKeys = extraInput.value.trim();
  }
  localStorage.setItem(STORAGE_KEY, JSON.stringify(config));
  log('配置及词库已保存到本地', 'ok');
}

  if (config.models && Array.isArray(config.models)) {
    config.models.forEach(m => {
      if (!models.find(x => x.id === m.id)) {
        models.push(m);
      }
    });
  }
  // 恢复活跃模型
  if (config.activeModelId && models.find(m => m.id === config.activeModelId)) {
    activeModelId = config.activeModelId;
  }
  // 恢复API Key
  setTimeout(() => {
    if (config.apiKeys) {
      const currentKey = config.apiKeys[activeModelId];
      if (currentKey) {
        $('apiKeyInput').value = currentKey;
      }
    }
    if (config.extraKeys) {
      const extraInput = $('extraKeysInput');
      if (extraInput) extraInput.value = config.extraKeys;
    }
  }, 100);

  // 恢复词库数据
  if (config.lexicon) {
    if (config.lexicon.entity !== undefined) $('entityTextarea').value = config.lexicon.entity;
    if (config.lexicon.domain !== undefined) $('domainTextarea').value = config.lexicon.domain;

  }
}

async function callLLM(model, apiKey, content, systemPrompt, maxRetries = 4) {
  let attempt = 0;
  while (true) {
    let res;
    try {
      // 增加简单的超时机制
      const controller = new AbortController();
      const timeoutId = setTimeout(() => controller.abort(), 15000); // 15秒超时
      
      res = await fetch(model.url + '/chat/completions', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json', 'Authorization': 'Bearer ' + apiKey },
        signal: controller.signal,
        body: JSON.stringify({
          model: model.modelId,
          messages: [
            { role: 'system', content: systemPrompt || '你是专业舆情分析员' },
            { role: 'user', content: content }
          ],
          max_tokens: 256,
          temperature: 0.1
        })
      });
      clearTimeout(timeoutId);
    } catch (netErr) {
      // 网络抖动或超时：可重试
      if (attempt < maxRetries) { await sleep(backoffMs(attempt)); attempt++; continue; }
      throw netErr;
    }
    // 限速 / 服务端临时错误 → 退避重试
    if ((res.status === 429 || res.status >= 500) && attempt < maxRetries) {
      const ra = parseInt(res.headers.get('retry-after') || '', 10);
      await sleep(Number.isFinite(ra) ? ra * 1000 : backoffMs(attempt));
      attempt++;
      continue;
    }
    if (!res.ok) {
      const txt = await res.text().catch(() => '');
      throw new Error(`HTTP ${res.status}: ${txt.slice(0, 100)}`);
    }
    const data = await res.json();
    return data.choices?.[0]?.message?.content || '';
  }
}

// 指数退避（带抖动）：0.8s, 1.6s, 3.2s, 6.4s …
function backoffMs(attempt) {
  return Math.round((0.8 * Math.pow(2, attempt)) * 1000 * (0.7 + Math.random() * 0.6));
}

//  本地评分核心

function buildEntitySet() { return new Set(getLexicon($('entityTextarea')).map(w => w.toLowerCase())); }
function buildDomainSet() { return new Set(getLexicon($('domainTextarea')).map(w => w.toLowerCase())); }
function buildNegSet() { return new Set(getLexicon($('negTextarea')).map(w => w.toLowerCase())); }

// 强实体：指向唯一、单独出现即可确认主体（含「国网/国家电网」或为具名子公司/省公司）
// 弱实体：泛化、单独出现不足以确认主体，需与领域词邻近共现
const STRONG_ENTITY_EXTRA = new Set([
  '南瑞','许继','平高','英大','新源','中电装备','山东电工','南瑞继保',
  '95598','网上国网'
].map(w => w.toLowerCase()));

// 用户自定义「强实体豁免词」：填入即强制判为强实体（绕过常规规则）
let userExemptSet = new Set();
function refreshExemptSet() {
  const ta = $('exemptTextarea');
  userExemptSet = new Set((ta ? getLexicon(ta) : []).map(w => w.toLowerCase()));
}

function isStrongEntity(word) {
  const w = word.toLowerCase();
  if (userExemptSet.has(w)) return true;                          // 用户强制豁免
  if (w.includes('国网') || w.includes('国家电网')) return true;   // 国网XX 系列
  if (STRONG_ENTITY_EXTRA.has(w)) return true;                     // 具名子公司
  if (w.length >= 5) return true;                                  // 足够长的具名机构
  return false;
}

function classifyEntities(entitySet) {
  const strong = [], weak = [];
  for (const e of entitySet) { (isStrongEntity(e) ? strong : weak).push(e); }
  return { strong, weak };
}

function containsDomain(text, domainSet) {
  if (!text) return false;
  const t = text.toLowerCase();
  for (const d of domainSet) { if (t.includes(d)) return true; }
  return false;
}

// 列出文本中命中的全部某类词（用于结果展示）
function listWordHits(text, wordList) {
  if (!text) return [];
  const lower = text.toLowerCase();
  const hits = [];
  for (const w of wordList) { if (w && lower.includes(w)) hits.push(w); }
  return [...new Set(hits)];
}

// 找出文本中所有某词的出现位置（区间 [start,end)）
function findOccurrences(lower, word) {
  const res = [];
  if (!word) return res;
  let from = 0, idx;
  while ((idx = lower.indexOf(word, from)) !== -1) {
    res.push([idx, idx + word.length]);
    from = idx + word.length;
  }
  return res;
}

// 实体门槛判定：
//  - 强实体在文中任意位置出现 → 通过（主体明确）
//  - 否则弱实体须与领域词「邻近共现」（间隔 ≤ proximity 字）→ 通过
//  proximity<=0 表示关闭邻近约束，退化为「弱实体只要出现即可」
function entityGate(text, strongList, weakList, domainList, proximity) {
  if (!text) return false;
  const lower = text.toLowerCase();

  // 1. 强实体直接放行
  for (const e of strongList) { if (lower.includes(e)) return true; }

  if (!weakList.length) return false;

  // 2. 弱实体：无邻近约束时，出现即可
  if (proximity <= 0) {
    for (const e of weakList) { if (lower.includes(e)) return true; }
    return false;
  }

  // 3. 弱实体 + 领域词 邻近共现
  const domainPos = [];
  for (const d of domainList) {
    for (const oc of findOccurrences(lower, d)) domainPos.push(oc);
  }
  if (!domainPos.length) return false;

  for (const e of weakList) {
    for (const [es, ee] of findOccurrences(lower, e)) {
      for (const [ds, de] of domainPos) {
        // 两区间最近边距
        const gap = es >= de ? es - de : (ds >= ee ? ds - ee : 0);
        if (gap <= proximity) return true;
      }
    }
  }
  return false;
}

// 返回 { score, hitWords }
function localScore(text, negSet) {
  if (!text) return { score: 0, hitWords: [] };
  const lower = text.toLowerCase();
  let score = 0;
  const hitWords = [];

  // 1. 复合短语（最高权重）
  for (const phrase of COMPOUND_NEG) {
    if (lower.includes(phrase.toLowerCase())) {
      score += 1.5;
      hitWords.push(phrase);
    }
  }

  // 2. 负面词基础匹配
  const parts = lower.split(/[\s,，。、；;！!？?<>《》()\[\]【】"""'''\n\r\t]+/).filter(w => w.length > 0);
  const usedWords = new Set();
  for (let i = 0; i < parts.length; i++) {
    const w = parts[i];
    if (negSet.has(w) && !usedWords.has(w)) {
      let weight = 1.0;
      // 强度副词前置
      if (i > 0 && ['非常','极其','严重','十分','特别','超级','巨'].includes(parts[i-1])) weight += 0.3;
      // 叠词强调
      if (i > 0 && parts[i-1] === w) weight = 1.8;
      score += weight;
      usedWords.add(w);
      hitWords.push(w);
    }
  }

  // 3. 子串兜底（针对词典中的长词组）
  for (const nw of negSet) {
    if (nw.length >= 2 && lower.includes(nw) && !usedWords.has(nw)) {
      score += 0.5;
      hitWords.push(nw);
    }
  }

  // 4. 否定词+否定组合额外加分
  const negations = ['没有','不会','无法','不能','不予'];
  for (const ng of negations) {
    if (lower.includes(ng)) { score += 0.3; break; }
  }

  // 5. 标点强化
  if (/[！!]{2,}|？！|！？/.test(text)) score += 0.2;

  return { score: Math.min(1.0, score / 2.5), hitWords: [...new Set(hitWords)] };
}

async function startAnalysis() {
  if (!wb) return;
  if (isRunning) return;
  isRunning = true;
  shouldStop = false;
  resultRows = [];

  // 释放上一次导出的 Blob 缓存，防止内存泄漏
  if (window._exportBlob) {
    window._exportBlob = null;
  }
  if (window._extractBlob) {
    window._extractBlob = null;
  }

  $('startBtn').disabled = true;
  $('stopBtn').style.display = '';
  $('progressWrap').style.display = '';
  $('topExportBtn').disabled = true;
  setStatus('识别中...', 'warn');

  // 切换到日志tab
  document.querySelectorAll('.main-tab').forEach(t => t.classList.remove('active'));
  document.querySelectorAll('.tab-pane').forEach(p => p.classList.remove('active'));
  document.querySelector('.main-tab[data-tab="log"]').classList.add('active');
  $('tab-log').classList.add('active');

  const start = Date.now();
  log('═══════════════════════════════', 'info');
  log(`开始识别 | 模式: ${currentMode === 'ai' ? 'AI模式（本地+LLM）' : '本地模式'} | Sheet: ${currentSheetName || wb.SheetNames[0]}`, 'info');

  const sheetName = currentSheetName || wb.SheetNames[0];
  const ws = wb.Sheets[sheetName];
  const rawData = XLSX.utils.sheet_to_json(ws, { header: 1, defval: '' });

  if (rawData.length < 2) {
    log('数据为空或只有标题行', 'err');
    finishAnalysis(start);
    return;
  }

  const headers = rawData[0].map(h => (h || '').toString().trim());
  const idxYq = headers.findIndex(h => h.includes('是否舆情') || h.includes('舆情标记'));

  // 检查用户是否在"字段映射"中进行了自定义选择
  const mappingSelects = document.querySelectorAll('.mapping-select');
  const customCols = [];
  mappingSelects.forEach(sel => {
    if (sel.value !== "") customCols.push(parseInt(sel.value, 10));
  });

  let targetCols = []; // 最终决定要分析的列集合

  if (customCols.length > 0) {
    // 方案 A：用户有自定义选择，严格遵循用户的选择
    log(`检测到自定义字段映射，已覆盖系统默认规则，将扫描指定的 ${customCols.length} 列`, 'info');
    targetCols = customCols.map(idx => ({ colIdx: idx, label: headers[idx] || `列${idx}` }));
  } else {
    // 方案 B：用户未选择，使用经典的默认探测逻辑
    const idxTitle = headers.findIndex(h => h.includes('标题') || h.includes('微博内容'));
    const idxAbs = headers.findIndex(h => h === '摘要');

    log(`默认列映射 → 标题列:${idxTitle >= 0 ? headers[idxTitle]+'('+idxTitle+')' : '未找到'} | 摘要列:${idxAbs >= 0 ? headers[idxAbs]+'('+idxAbs+')' : '未找到'} | 结果列:${idxYq >= 0 ? headers[idxYq]+'('+idxYq+')' : '默认 B 列'}`, 'info');

    if (idxTitle < 0 && idxAbs < 0) {
      log('未找到默认分析列（标题/摘要），请去「字段映射」中手动指定要分析的列。', 'err');
      finishAnalysis(start);
      return;
    }

    targetCols = [
      { colIdx: idxTitle, label: '标题' },
      { colIdx: idxAbs, label: '摘要' }
    ];
  }

  const requireEntity = $('requireEntity').checked;
  const threshold = parseFloat($('thresholdSlider').value);
  refreshExemptSet();
  const entitySet = buildEntitySet();
  const domainSet = buildDomainSet();
  const negSet = buildNegSet();

  // 实体词分层 + 邻近共现距离
  const { strong: strongEntities, weak: weakEntities } = classifyEntities(entitySet);
  const domainList = [...domainSet];
  const proxEl = $('proximityInput');
  const proximity = proxEl ? parseInt(proxEl.value, 10) || 0 : 30;

  log(`词库规模 → 实体词: ${entitySet.size}（强 ${strongEntities.length} / 弱 ${weakEntities.length}） | 领域词: ${domainSet.size} | 负面词: ${negSet.size}`, 'info');
  log(`阈值: ${threshold.toFixed(2)} | 实体词过滤: ${requireEntity ? '开启' : '关闭'} | 弱实体邻近共现距离: ${proximity > 0 ? proximity + '字' : '关闭'}`, 'info');

  const totalRows = rawData.length - 1;
  let localMarked = 0;

  // ── 本地模式扫描 ──
  log(`开始本地扫描，共 ${totalRows} 行...`, 'info');

  // 每行的命中情况记录，用于后续写xlsx
  // hitInfo[rowIdx] = { yqYes: bool, cells: [{colIdx, hitWords, entityHits, domainHits}] }
  const hitInfo = new Map();

  const BATCH = 200;
  for (let r = 1; r < rawData.length; r++) {
    if (shouldStop) break;

    // 进度
    if (r % BATCH === 0 || r === rawData.length - 1) {
      const pct = Math.round((r / (rawData.length - 1)) * 100);
      $('progressFill').style.width = pct + '%';
      $('progressInfo').textContent = `本地扫描 ${r} / ${totalRows}`;
      $('progressPct').textContent = pct + '%';
      $('statTotal').textContent = r;
      await sleep(0);
    }

    const row = rawData[r];

    let rowHit = false;
    const cellHits = [];

    // 逐字段独立判断（兼容默认与自定义）
    for (const target of targetCols) {
      const colIdx = target.colIdx;
      if (colIdx < 0 || colIdx === undefined) continue;

      const text = String(row[colIdx] || '');
      const label = target.label;
      if (!text) continue;

      // 实体词门槛：强实体直接放行；弱实体须与领域词邻近共现
      if (requireEntity && !entityGate(text, strongEntities, weakEntities, domainList, proximity)) continue;
      // 领域词门槛
      if (!containsDomain(text, domainSet)) continue;

      const { score, hitWords } = localScore(text, negSet);
      if (score >= threshold && hitWords.length > 0) {
        rowHit = true;
        const entityHits = listWordHits(text, [...strongEntities, ...weakEntities]);
        const domainHits = listWordHits(text, domainList);
        cellHits.push({ colIdx, hitWords, label, entityHits, domainHits });
      }
    }

    if (rowHit) {
      localMarked++;
      hitInfo.set(r, { yqYes: true, cells: cellHits });
      resultRows.push({ rowIdx: r, rowData: row, hitCells: cellHits, aiConfirmed: false, headers });
      $('statMarked').textContent = localMarked;
    }
  }

  log(`本地扫描完成，共标记 ${localMarked} 行`, localMarked > 0 ? 'warn' : 'ok');

  // ── AI 二次核验（并发池 + 多 Key 轮询）──
  let aiMarked = 0;
  if (currentMode === 'ai' && !shouldStop) {
    const primaryKey = $('apiKeyInput').value.trim();
    const extraKeys = ($('extraKeysInput') ? $('extraKeysInput').value : '')
      .split('\n').map(s => s.trim()).filter(Boolean);
    const keys = [...new Set([primaryKey, ...extraKeys].filter(Boolean))];
    const model = models.find(m => m.id === activeModelId);
    let concurrency = parseInt(($('concurrencyInput') || {}).value, 10);
    if (!Number.isFinite(concurrency) || concurrency < 1) concurrency = 8;
    concurrency = Math.min(32, concurrency);

    if (!keys.length || !model) {
      log('AI模式：未配置 API Key 或模型，跳过 LLM 核验', 'warn');
    } else {
      const aiTotal = resultRows.length;
      log(`AI核验开始 → ${aiTotal} 条 | 并发 ${concurrency} | Key 数 ${keys.length}${keys.length > 1 ? '（轮询）' : ''}`, 'info');

      const systemP = `你是一名专业的企业舆情分析员，专门识别与国家电网公司招聘和薪酬管理相关的负面信息。\n只输出 JSON，不要输出任何其他内容。`;
      let done = 0;
      let nextIdx = 0;        // 任务游标
      let keyCursor = 0;      // Key 轮询游标

      // 单个任务：核验第 i 条
      async function processOne(i) {
        const rr = resultRows[i];
        const row = rr.rowData;

        try {
          // 修复：动态遍历 targetCols（尊重用户的自定义字段映射）
          let combinedContent = '';
          for (const target of targetCols) {
            if (target.colIdx >= 0) {
              const text = String(row[target.colIdx] || '').trim();
              if (text) combinedContent += `【${target.label}】：${text}\n`;
            }
          }

          // 安全截断，防止超出模型上下文限制
          if (combinedContent.length > 1500) {
            combinedContent = combinedContent.substring(0, 1500) + '...(已截断)';
          }

          const prompt = buildAIPrompt(combinedContent);
          const key = keys[(keyCursor++) % keys.length];   // 轮询分配 Key

          const reply = await callLLM(model, key, prompt, systemP);
          const json = parseAIReply(reply);

          if (json && (json.r === true || json.r === "true" || json.overall_negative === true)) {
            rr.aiConfirmed = true;
            aiMarked++;
            $('statAI').textContent = aiMarked;
          }
        } catch (e) {
          log(`AI核验第${i + 1}条失败: ${e.message}，保留本地标记`, 'warn');
        } finally {
          done++;
          if (done % 5 === 0 || done === aiTotal) {
            const pct = Math.round((done / aiTotal) * 100);
            $('progressFill').style.width = pct + '%';
            $('progressInfo').textContent = `AI 核验 ${done} / ${aiTotal}`;
            $('progressPct').textContent = pct + '%';
          }
        }
      }

      // 并发 worker：持续领取下一条，直到取完或被中止
      async function worker() {
        while (true) {
          if (shouldStop) return;
          const i = nextIdx++;
          if (i >= aiTotal) return;
          await processOne(i);
        }
      }

      const workers = [];
      for (let w = 0; w < Math.min(concurrency, aiTotal); w++) workers.push(worker());
      await Promise.all(workers);

      log(`AI核验完成，确认负面: ${aiMarked} 条`, aiMarked > 0 ? 'warn' : 'ok');
    }
  }

  // ── 写入 xlsx（使用 ExcelJS 实现三类关键词对应颜色）──
  log('开始生成标注文件...', 'info');

  // 保持进度条面板显示
  $('progressWrap').style.display = '';
  $('stopBtn').style.display = 'none'; // 导出阶段不可中止

  try {
    // 传入进度回调函数，实时更新 UI
    window._exportBlob = await buildOutputWorkbookEJS(
      idxYq,
      hitInfo,
      resultRows,
      currentMode === 'ai',
      (msg, pct) => {
        $('progressFill').style.width = pct + '%';
        $('progressInfo').textContent = msg;
        $('progressPct').textContent = pct + '%';
      }
    );
    window._exportWb = null;
  } catch (e) {
    log('标注文件生成失败: ' + e.message, 'err');
    window._exportBlob = null;
  }
  window._exportFileName = originalFileName + '-识别结果.xlsx';

  $('topExportBtn').disabled = false;

  // 更新统计
  const elapsed = ((Date.now() - start) / 1000).toFixed(1);
  $('statTime').textContent = elapsed + 's';
  $('statTotal').textContent = totalRows;

  log(`识别完成 ✓ | 总行: ${totalRows} | 本地标记: ${localMarked} | AI确认: ${aiMarked} | 耗时: ${elapsed}s`, 'ok');
  log('═══════════════════════════════', 'info');

  // 切换到结果tab
  document.querySelectorAll('.main-tab').forEach(t => t.classList.remove('active'));
  document.querySelectorAll('.tab-pane').forEach(p => p.classList.remove('active'));
  document.querySelector('.main-tab[data-tab="result"]').classList.add('active');
  $('tab-result').classList.add('active');

  renderPreview(resultRows);
  $('resultCount').textContent = localMarked + ' 条本地标记';
  if (currentMode === 'ai') {
    $('resultAICount').style.display = '';
    $('resultAICount').textContent = aiMarked + ' 条 AI 确认';
  }

  setStatus(`完成 | ${localMarked} 条负面`, 'success');
  finishAnalysis(start);
}

function buildAIPrompt(content) {

function parseAIReply(reply) {
  try {
    const clean = reply.replace(/```json|```/g, '').trim();
    const m = clean.match(/\{[\s\S]*\}/);
    if (!m) return null;
    return JSON.parse(m[0]);
  } catch { return null; }
}

// ═══════════════════════════════════════════
//  关键词级富文本：支持三类词（实体蓝、领域紫、情感红）着色
// ═══════════════════════════════════════════
function buildRichText(text, entityHits, domainHits, negHits) {
  if (!text) return null;
  
  // 按照 UI 中的颜色设置对应的 ARGB 
  // 实体(蓝): #60a5fa -> FF60A5FA
  // 领域(紫): #a78bfa -> FFA78BFA
  // 情感(红): #ef4444 -> FFEF4444
  const wordCategories = [
    { words: [...new Set(entityHits || [])].filter(Boolean), argb: 'FF60A5FA' },
    { words: [...new Set(domainHits || [])].filter(Boolean), argb: 'FFA78BFA' },
    { words: [...new Set(negHits || [])].filter(Boolean), argb: 'FFEF4444' }
  ];
  
  // 验证是否有命中的词
  if (!wordCategories.some(cat => cat.words.length > 0)) return null;
  
  const lower = text.toLowerCase();
  
  // 使用与原文本等长的数组来记录每个字符的颜色，默认为 null（无色）
  const charColors = new Array(text.length).fill(null);
  
  // 依次应用染色规则。因为是依次覆盖，所以排在后面的优先级更高。
  // 设定优先级：情感(红) > 领域(紫) > 实体(蓝)
  for (const cat of wordCategories) {
    // 对命中词按长度降序排列，确保长词优先整体匹配（如果在同一类别中）
    const words = cat.words.sort((a, b) => b.length - a.length);
    for (const w of words) {
      const wl = w.toLowerCase();
      if (!wl) continue;
      let from = 0, idx;
      while ((idx = lower.indexOf(wl, from)) !== -1) {
        // 给匹配到的区间字符上色
        for (let k = idx; k < idx + wl.length; k++) {
          charColors[k] = cat.argb;
        }
        from = idx + wl.length;
      }
    }
  }
  
  // 如果没有任何字符被上色，直接返回 null
  if (!charColors.some(Boolean)) return null; 
  
  const runs = [];
  let i = 0;
  while (i < text.length) {
    const color = charColors[i];
    let j = i;
    // 聚合颜色相同的连续字符
    while (j < text.length && charColors[j] === color) j++;
    const seg = text.slice(i, j);
    
    if (color) {
      runs.push({ text: seg, font: { color: { argb: color }, name: 'Microsoft YaHei' } });
    } else {
      runs.push({ text: seg, font: { name: 'Microsoft YaHei' } });
    }
    
    i = j;
  }
  
  return { richText: runs };
}

// ═══════════════════════════════════════════
//  生成输出 Workbook（优化版：时间分片防假死）
//  返回 Blob，供 doExport 直接下载
// ═══════════════════════════════════════════
async function buildOutputWorkbookEJS(idxYq, hitInfo, aiRows, isAIMode, onProgress) {
  if (!window._origArrayBuffer) throw new Error('原始文件缓存丢失，请重新上传');

  const sheetName = currentSheetName || wb.SheetNames[0];
  const yqColIdx = idxYq >= 0 ? idxYq : 1;
  const yqColEJS = yqColIdx + 1;

  const aiConfirmedSet = new Set(aiRows.filter(r => r.aiConfirmed).map(r => r.rowIdx));
  const cellsByRow = new Map();
  for (const [rowIdx, info] of hitInfo) cellsByRow.set(rowIdx, info.cells);

  if (onProgress) onProgress('正在加载原始文档...', 10);
  const ejsWb = new ExcelJS.Workbook();
  await ejsWb.xlsx.load(window._origArrayBuffer.slice(0));

  let ws = ejsWb.getWorksheet(sheetName) || ejsWb.worksheets[0];
  if (!ws) throw new Error('未找到工作表');

  if (onProgress) onProgress('正在渲染高亮结果...', 30);

  let processedCount = 0;
  const totalHits = cellsByRow.size;

  for (const [rowIdx, cells] of cellsByRow) {
    const excelRow = rowIdx + 1;
    const isAIConfirmed = aiConfirmedSet.has(rowIdx);

    // B 列写"是"：本地=黑色；AI确认=红色加粗
    const yqCell = ws.getCell(excelRow, yqColEJS);
    yqCell.value = '是';
    yqCell.font = {
      name: 'Microsoft YaHei',
      color: { argb: isAIConfirmed ? 'FFFF0000' : 'FF000000' },
      bold: !!isAIConfirmed
    };

    // 命中单元格：命中关键词三类分别着色
    for (const { colIdx, hitWords, entityHits, domainHits } of cells) {
      if (colIdx < 0) continue;
      const cell = ws.getCell(excelRow, colIdx + 1);
      // 跳过公式单元格，避免破坏底层 XML 结构
      if (cell.formula) continue;
      const text = (cell.text != null ? String(cell.text) : (cell.value != null ? String(cell.value) : ''));
      const rt = buildRichText(text, entityHits, domainHits, hitWords);
      if (rt) cell.value = rt;
    }

    processedCount++;
    // 核心优化：每渲染 50 行，主动释放一次主线程，避免 UI 卡死
    if (processedCount % 50 === 0) {
      if (onProgress) {
        // 估算渲染进度 (30% ~ 70% 之间)
        const renderPct = 30 + Math.round((processedCount / totalHits) * 40);
        onProgress(`正在渲染高亮... ${processedCount}/${totalHits}`, renderPct);
      }
      await sleep(0);
    }
  }

  if (onProgress) onProgress('正在封装导出文件 (这可能需要几秒钟)...', 80);

  // 留出时间让浏览器渲染上一句提示
  await sleep(50);

  const buf = await ejsWb.xlsx.writeBuffer();
  if (onProgress) onProgress('文件生成完毕', 100);

  return new Blob([buf], { type: 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet' });
}

function renderPreview(rows) {
  const wrap = $('previewWrap');
  if (!rows.length) {
    wrap.innerHTML = `<div class="empty-state"><div class="emoji">🔎</div><p>识别完成后，结果将展示在这里<br>仅显示被标记为负面舆情的数据行</p></div>`;
    return;
  }
  const headers = rows[0].headers;

  // 展示关键列：行号、是否舆情、标题、摘要、命中词、AI确认
  const cols = ['行号'];
  const titleLabel = headers.find(h => h.includes('标题') || h.includes('微博内容')) || '标题';
  const absLabel = headers.find(h => h === '摘要') || '摘要';
  cols.push(titleLabel, absLabel, '命中词（实体/领域/情感）', 'AI 确认');

  let html = `<div class="preview-wrap"><table class="preview-table"><thead><tr>`;
  cols.forEach(c => { html += `<th>${escHtml(c)}</th>`; });
  html += `</tr></thead><tbody>`;

  for (const rr of rows) {
    const row = rr.rowData;
    const idxTitle = rr.headers.findIndex(h => h.includes('标题') || h.includes('微博内容'));
    const idxAbs = rr.headers.findIndex(h => h === '摘要');
    const entityHits = [...new Set(rr.hitCells.flatMap(c => c.entityHits || []))];
    const domainHits = [...new Set(rr.hitCells.flatMap(c => c.domainHits || []))];
    const negHits = [...new Set(rr.hitCells.flatMap(c => c.hitWords || []))];

    const chip = (w, color) => `<span style="display:inline-block;padding:1px 7px;margin:2px 3px 2px 0;border-radius:6px;font-size:11px;background:${color}1a;color:${color};border:1px solid ${color}55;">${escHtml(w)}</span>`;
    let kwHtml = '';
    if (entityHits.length) kwHtml += `<div style="margin-bottom:2px;"><span style="font-size:10px;color:var(--text3);">实体</span> ${entityHits.map(w => chip(w, '#60a5fa')).join('')}</div>`;
    if (domainHits.length) kwHtml += `<div style="margin-bottom:2px;"><span style="font-size:10px;color:var(--text3);">领域</span> ${domainHits.map(w => chip(w, '#a78bfa')).join('')}</div>`;
    if (negHits.length) kwHtml += `<div><span style="font-size:10px;color:var(--text3);">情感</span> ${negHits.map(w => chip(w, '#ef4444')).join('')}</div>`;
    if (!kwHtml) kwHtml = '—';

    html += `<tr class="row-marked">
      <td>${rr.rowIdx}</td>
      <td class="hit" title="${escHtml(String(row[idxTitle] || ''))}">${escHtml(String(row[idxTitle] || '').slice(0, 50))}${(row[idxTitle]||'').length > 50 ? '…' : ''}</td>
      <td class="hit" title="${escHtml(String(row[idxAbs] || ''))}">${escHtml(String(row[idxAbs] || '').slice(0, 50))}${(row[idxAbs]||'').length > 50 ? '…' : ''}</td>
      <td style="min-width:220px;">${kwHtml}</td>
      <td class="${rr.aiConfirmed ? 'yq-ai' : ''}">${rr.aiConfirmed ? '✓ 已确认' : '—'}</td>
    </tr>`;
  }
  html += '</tbody></table></div>';
  wrap.innerHTML = html;
}

    log(`文件已导出: ${fileName}`, 'ok');

    log('导出失败: ' + e.message, 'err');
  }
}

  const colIdx = parseInt($('extractColSelect').value, 10);
  if (isNaN(colIdx)) { alert('请选择包含 URL 的列'); return; }

  const concurrency = parseInt($('extractConcurrency').value, 10) || 20;
  const sheetName = currentSheetName || wb.SheetNames[0];
  const ws = wb.Sheets[sheetName];
  const rawData = XLSX.utils.sheet_to_json(ws, { header: 1, defval: '' });

  if (rawData.length < 2) { alert('数据为空'); return; }

  const urlTasks = [];
  for (let r = 1; r < rawData.length; r++) {
    const cellText = String(rawData[r][colIdx] || '').trim();
    const urlMatch = cellText.match(/https?:\/\/[^\s<>"]+/);
    if (urlMatch) {
      urlTasks.push({ rowIdx: r, url: urlMatch[0] });
    }
  }

  if (!urlTasks.length) { alert('所选列中未找到 URL'); return; }

  $('extractStartBtn').disabled = true;
  $('extractProgressWrap').style.display = '';
  $('extractStatSuccess').textContent = '0';
  $('extractStatLogin').textContent = '0';
  $('extractStatFail').textContent = '0';

  const total = urlTasks.length;
  let done = 0, success = 0, login = 0, fail = 0;

  log(`内容解析开始 → ${total} 条 URL | 并发 ${concurrency}`, 'info');

  try {
    const results = await window.electronAPI.extractUrls(urlTasks, concurrency, (progress) => {
      // 需要登录的通知
      if (progress.needLogin) {
        log(`需要登录: ${progress.domain}`, 'warn');
        showLoginConfirm(progress.domain);
        return;
      }

      // 普通消息
      if (progress.message) {
        log(progress.message, 'warn');
        return;
      }

      // 实际结果
      if (progress.rowIdx !== undefined && done < total) {
        done++;
        if (progress.status === 'success') success++;
        else if (progress.status === 'login') login++;
        else fail++;

        const pct = Math.min(100, Math.round((done / total) * 100));
        $('extractProgressFill').style.width = pct + '%';
        $('extractProgressInfo').textContent = `解析中 ${done} / ${total}`;
        $('extractProgressPct').textContent = pct + '%';
        $('extractStatSuccess').textContent = success;
        $('extractStatLogin').textContent = login;
        $('extractStatFail').textContent = fail;
      }
    });

    log(`内容解析完成 → 成功: ${success} | 需登录: ${login} | 失败: ${fail}`, 'ok');
    writeExtractResults(results);

  } catch (e) {
    log('内容解析失败: ' + e.message, 'err');
  }

  $('extractStartBtn').disabled = false;
  log('内容解析完成 ✓', 'ok');
});

// 将提取结果写入 Excel（使用 ExcelJS 保留原格式）
async function writeExtractResults(results) {
  if (!results || !results.length) { log('没有提取结果可写入', 'warn'); return; }
  if (!window._origArrayBuffer) { log('原始文件缓存丢失，无法写入', 'err'); return; }

  // 调试日志
  const successCount = results.filter(r => r.status === 'success' && r.text).length;
  log(`准备写入 ${results.length} 条结果（其中成功提取 ${successCount} 条）`, 'info');

  try {
    const sheetName = currentSheetName || wb.SheetNames[0];

    // 使用 ExcelJS 读取原始文件，保留所有格式
    const ejsWb = new ExcelJS.Workbook();
    await ejsWb.xlsx.load(window._origArrayBuffer.slice(0));

    let ws = ejsWb.getWorksheet(sheetName) || ejsWb.worksheets[0];
    if (!ws) { log('未找到工作表', 'err'); return; }

    // 获取最大列数
    let maxCol = 0;
    ws.eachRow((row) => {
      let lastCol = 0;
      row.eachCell({ includeEmpty: false }, (cell, colNumber) => { lastCol = colNumber; });
      if (lastCol > maxCol) maxCol = lastCol;
    });

    const writeCol = maxCol + 1; // ExcelJS 1-based

    // 写入表头
    const headerCell = ws.getCell(1, writeCol);
    headerCell.value = '抓取原文';
    headerCell.font = { name: 'Microsoft YaHei', bold: true };

    // 写入提取结果
    let writtenCount = 0;
    for (const result of results) {
      const rowNum = result.rowIdx + 1; // rowIdx 是 0-based，ExcelJS 是 1-based
      const cell = ws.getCell(rowNum, writeCol);
      const textToWrite = result.text || '提取失败';
      cell.value = textToWrite;
      cell.font = { name: 'Microsoft YaHei' };
      writtenCount++;

      // 调试：前3条记录详细日志
      if (writtenCount <= 3) {
        log(`写入行${rowNum}列${writeCol}: "${textToWrite.substring(0, 30)}..."`, 'info');
      }
    }

    // 生成独立的导出 Blob
    const buf = await ejsWb.xlsx.writeBuffer();
    window._extractBlob = new Blob([buf], { type: 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet' });
    window._extractFileName = (originalFileName || '数据') + '-含抓取原文.xlsx';

    log(`已将 ${writtenCount} 条结果写入「抓取原文」列（第 ${writeCol} 列）`, 'ok');
    log('含抓取原文的文件已生成，请点击"导出结果"下载', 'ok');

    // 启用导出按钮
    $('topExportBtn').disabled = false;

  } catch (e) {
    log('写入提取结果失败: ' + e.message, 'err');
  }
}
